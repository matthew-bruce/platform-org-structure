```typescript
async function duplicatePerson(person: Person, targetTeamId: string) {
  const targetTeam = teams.find(t => t.id === targetTeamId)
  if (!targetTeam) return
  
  const duplicate = {
    ...person,
    id: undefined,
    team_id: targetTeamId,
    tower_id: targetTeam.tower_id,
    capacity: 50,
    sort_order: 999,
    created_at: undefined,
    updated_at: undefined
  }
  
  const { error } = await supabase
    .from('people')
    .insert([duplicate])
  
  if (error) {
    console.error('Error duplicating person:', error)
    alert('Failed to duplicate person')
  } else {
    fetchPeople()
  }
}
```

**UPDATE `handlePersonDoubleClick`** to edit multiple fields:
```typescript
function handlePersonDoubleClick(person: Person) {
  // Create modal or prompt for each field
  const newName = prompt('Name:', person.name)
  if (newName === null) return
  
  const newTitle = prompt('Title:', person.title)
  if (newTitle === null) return
  
  const updates: Partial<Person> = { ...person }
  if (newName) updates.name = newName
  if (newTitle) updates.title = newTitle
  
  if (isAdminMode) {
    const newRate = prompt('Commercial Rate (£/day):', person.commercial_rate.toString())
    if (newRate) updates.commercial_rate = parseInt(newRate)
  }
  
  const newCapacity = prompt('Capacity (%):', person.capacity.toString())
  if (newCapacity) updates.capacity = parseInt(newCapacity)
  
  const newPlanview = prompt('Planview (BAU/F_GOV/PR):', person.planview)
  if (newPlanview && ['BAU', 'F_GOV', 'PR'].includes(newPlanview)) {
    updates.planview = newPlanview as any
  }
  
  updatePerson(updates as Person)
}
```
